<!-- Push Notification Logic -->
<script>
  // Show popup after 5 seconds if user hasn't subscribed
  setTimeout(function() {
    // If notifications are not supported or permission is already blocked/denied, do not show the popup
    if (!('Notification' in window) || Notification.permission === 'denied') {
      console.log('Push notifications not supported or permission blocked. Hiding popup.');
      return;
    }

    // Auto-sync localStorage if permission is already granted
    if (Notification.permission === 'granted') {
      localStorage.setItem('pushNotificationSubscribed', 'true');
      return;
    }

    const isSubscribed = localStorage.getItem('pushNotificationSubscribed');
    
    // Check if the popup was snoozed/dismissed within the last 24 hours
    const dismissedTime = localStorage.getItem('pushPopupDismissedTime');
    let isDismissed = false;
    if (dismissedTime) {
      const hoursPassed = (Date.now() - parseInt(dismissedTime, 10)) / (1000 * 60 * 60);
      if (hoursPassed < 24) {
        isDismissed = true;
      }
    }
    
    if (!isSubscribed && !isDismissed) {
      document.getElementById('pushNotificationPopup').classList.remove('hidden');
      console.log('🔔 Push notification popup shown');
    }
  }, 5000); // Show after 5 seconds

  // Allow notifications - called when user clicks "Allow" button
  async function allowPushNotifications() {
    console.log('✅ User clicked Allow button');
    document.getElementById('pushNotificationPopup').classList.add('hidden');
    
    // Call the subscription function from push-notification-helper.js
    if (window.PushNotifications && window.PushNotifications.subscribe) {
      await window.PushNotifications.subscribe();
    } else {
      console.error('❌ Push notification helper not loaded');
      alert('Error: Push notification system not ready. Please refresh the page.');
    }
  }

  // Dismiss popup for later - called when user clicks "Maybe Later"
  function laterPushNotifications() {
    console.log('⏸️ User clicked Maybe Later');
    document.getElementById('pushNotificationPopup').classList.add('hidden');
    localStorage.setItem('pushPopupDismissedTime', Date.now().toString());
  }
</script>
